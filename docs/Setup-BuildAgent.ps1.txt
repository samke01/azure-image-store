$AllProtocols = [System.Net.SecurityProtocolType]'Tls11,Tls12'
[System.Net.ServicePointManager]::SecurityProtocol = $AllProtocols

$workingdirectory = "C:\temp"
if ([System.IO.Directory]::Exists($workingdirectory) -eq $false ) {
    [System.IO.Directory]::CreateDirectory($workingdirectory) | Out-Null
}

foreach($sw in $software) {
    if( $sw.install -eq 1 ) {
        $swinstallDirectory = $sw.installDirectory
        if( [string]::IsNullOrEmpty($sw.version) -eq $false ) {
            $swversion = $sw.version.Replace(".", "_")
            $swinstallDirectory = [System.IO.Path]::Combine($sw.installDirectory, $swversion)
        }

        if ([System.IO.Directory]::Exists($swinstallDirectory) -eq $false ) {
            New-Item -Path $swinstallDirectory -ItemType Directory -force -Confirm:$false | Out-Null
        }

        $tfPath = [System.IO.Path]::Combine($swinstallDirectory, $sw.targetFile);
        
        if ( [System.IO.File]::Exists($tfPath) -eq $false ) {
            Invoke-WebRequest -Uri "$($sw.url)" -OutFile "${tfPath}" 

            if([System.IO.Path]::GetExtension($tfPath) -eq ".zip" ) {
                Expand-Archive "$tfPath" -DestinationPath "${targetDirectory}" -Force;
            }
        }

        if( $env:Path.Contains($swinstallDirectory) -eq $false -and $sw.setPath -eq 1 ) {
            if( $env:Path.EndsWith(";") -eq $false ) {
                $env:Path += ";"
            }
            $env:Path += "${swInstallDirectory};"
            [Environment]::SetEnvironmentVariable("Path", $env:Path, [System.EnvironmentVariableTarget]::Machine)
        }

        if( $sw.installType -eq "msi") {
            $execFile = $tfPath        
            $installedFile = $tfPath + ".txt"
            
            if( [System.IO.File]::Exists($installedFile) -eq $false ) {
                $file = Get-Item "$execFile"
            
                $MSIArguments = @(
                    "/i"
                    ('"{0}"' -f $file.FullName)
                    "/qn"
                    "/norestart"
                )
            
                Start-Process "msiexec.exe" -ArgumentList $MSIArguments
            
                "$installedFile" | out-file -FilePath "$installedFile" -Encoding UTF8
            }
        }

        if( $sw.installType -eq "devopsagent") {
            $configureFile = $sw.configureFile
            $installedFile = $sw.installedFile

            if( [System.IO.File]::Exists($installedFile) -eq $false ) {
                "${configureFile} --unattended --url $($sw.organisation) --agent ${env:computername}$($sw.agentsuffix) --pool $($sw.Pool) --work $($sw.Work) --token $($sw.Token) --runAsService --auth pat"
            
                "$installedFile" | out-file -FilePath "$installedFile" -Encoding UTF8
            }
        }
    }
}
